Application
  • Defines the namespace
  • Contains the ember components
  • Controls the lifecycle

Definition

            
                var App = Ember.Application.create();

Initialization

            
                App.ready = function() {
                    /** INIT CODE GOES HERE **/
                }
                App.initialize();
Model
  • Represent the objects the app manages
  • The properties are used for bindings in the template

Definition

            
                App.Movie = Ember.Object.extend({
                    title : null,
                    year: null,
                    rating: 0,
                    available: null,
                    imdbLink: null
                });

Usage

            
                var movie = App.Movie.create();
                movie.set('title','The Shawshank Redemption');
                movie.set('year', 1994);
                movie.set('rating', 9.2);
                movie.set('available', true);
                movie.set('imdbLink', 'http://www.imdb.com/title/tt0111161/');
